cmake_minimum_required(VERSION "3.30.0")

project (
    TextEditor
    VERSION 1.0
    DESCRIPTION "text editor" 
    LANGUAGES C CXX
)

# variable to hole path to sdl library
set(SDL_PATH ${CMAKE_CURRENT_SOURCE_DIR}/dependencies/lib)

# add sdl library 
add_library(SDL2 SHARED IMPORTED)

# tell location of dll and lib also the header file that sdl needs
set_target_properties(
    SDL2 PROPERTIES
    IMPORTED_LOCATION "${SDL_PATH}/SDL2.dll"
    IMPORTED_IMPLIB "${SDL_PATH}/SDL2.lib"
    INTERFACE_INCLUDE_DIRECTORIES "${SDL_PATH}/include"
)

add_executable(TextEditor main.c)

target_link_libraries(TextEditor PRIVATE SDL2)
include(CTest)

enable_testing()
